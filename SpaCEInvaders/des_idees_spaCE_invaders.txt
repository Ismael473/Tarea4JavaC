		// Requires at least one Alien to exist.
		// offset must start at 0.
		Alien GetRightmostAlien(offset)
			if !aliens[10-offset].dead
				return aliens[10-offset]
			if !aliens[21-offset].dead
				return aliens[21-offset]
			if !aliens[32-offset].dead
				return aliens[32-offset]
			if !aliens[43-offset].dead
				return aliens[43-offset]
			if !aliens[54-offset].dead
				return aliens[54-offset]
			return GetRightmostAlien(offset+1)

		// Similar to GetRightmostAlien()
		Alien GetLeftmostAlien(offset)
			if !aliens[0+offset].dead
				return aliens[0+offset]
			if !aliens[11+offset].dead
				return aliens[11+offset]
			if !aliens[22+offset].dead
				return aliens[22+offset]
			if !aliens[33+offset].dead
				return aliens[33+offset]
			if !aliens[44+offset].dead
				return aliens[44+offset]
			return GetRightmostAlien(offset+1)
